  function init_pagenation() {
    var pagescount, i, txt = "", activePage, fromPos, toPos, x, prevPage, nextPage, able = "",abler = "";
    pagescount = 13;
    activePage = window.location.pathname;
    if (activePage.indexOf(".html") == -1) { activePage = "/index.html"; }
    if (activePage.indexOf("index.html") > -1) {
        activePage = 1;
    } else {
        fromPos = activePage.indexOf("tutorial_page") + 13; //length of tutorial_page
        toPos = activePage.indexOf(".html");
        activePage = activePage.substring(fromPos, toPos);
    }
    if (activePage == 1) {
        able = " pagdisabled"
        prevPage = "index.html"
    } else if (activePage == 2) {
        prevPage = "index.html";
    } else {
        prevPage = "tutorial_page" + (activePage - 1) + ".html";
    }
    if (activePage == pagescount) {
        nextPage = "tutorial_page13.html"; 
        abler = " pagdisabled"
    } else {
        nextPage = "tutorial_page" + (parseInt(activePage) + 1) + ".html";
    }
    x = document.getElementsByClassName("pagenation");
    for (i = 0; i < x.length; i++) {
        x[i].innerHTML = txt;
        x[i].getElementsByClassName("pagenation_item")[activePage].classList.add("pagactive");
    }
}
init_pagenation();
